Javascript Vanilla!

Incluindo JS no arquivo HTML

  • Inline
  • Interno
  • Externo

Inline:

      
        <button onclick="alert('Olá, mundo!')">Clique em mim</button>

Interno:

      
        <script>
          alert('Olá, mundo!');
        </script>

Externo:

      
        <script src="script.js"></script>

Selecionando Elementos com JS

Por ID

    
      document.getElementById('id-do-elemento');

Por Tag

    
      document.getElementsByTagName('tag-do-elemento');

Por Classe

    
      document.getElementsByClassName('classe-do-elemento');

Por Seletor

    
      document.querySelector('seletor-do-elemento');
      document.querySelector('#menuPrincipal')
      document.querySelector('div.card')
      document.querySelector('div p.lead')

Criando novos elementos

    
      const novoElemento = document.createElement('tag-do-elemento');

Adicionando elementos ao DOM

    
      const novoElemento = document.createElement('tag-do-elemento');
      document.body.appendChild(novoElemento);

Removendo elementos do DOM

   
     const elemento = document.getElementById('id-do-elemento');
     document.body.removeChild(elemento);

Manipulando o conteúdo e atributos dos elementos HTML

    
      const elemento = document.getElementById('id-do-elemento');
      elemento.innerText = 'Novo conteúdo';
    
    
      const elemento = document.getElementById('id-do-elemento');
      elemento.innerHTML = 'Novo conteúdo';
    
    
      const elemento = document.getElementById('id-do-elemento');
      elemento.setAttribute('atributo', 'valor');
    
    
      const elemento = document.getElementById('id-do-elemento');
      elemento.removeAttribute('atributo');

Manipulando o estilo dos elementos

    
      const elemento = document.getElementById('id-do-elemento');
      elemento.style.backgroundColor = 'red';
    
    
      const elemento = document.getElementById('id-do-elemento');
      elemento.style.fontSize = '2rem';

Alterando classes CSS

    
      const elemento = document.getElementById('id-do-elemento');
      elemento.classList.add('nova-classe');
    
    
      const elemento = document.getElementById('id-do-elemento');
      elemento.classList.remove('classe-a-remover');
    
    
      const elemento = document.getElementById('id-do-elemento');
      elemento.classList.toggle('classe-a-alternar');

Eventos

    
      const botao = document.getElementById('id-do-botao');
      botao.addEventListener('click', function() {
        alert('Clicou no botão!');
      });
    
  
    
      const botao = document.getElementById('id-do-botao');
      botao.addEventListener('click', function() {
        alert('Clicou no botão!');
      });
